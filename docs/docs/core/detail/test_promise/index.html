实现一个模拟变化接口的 promise 函数

每次调用都返回不同的结果#

const testPromiseFn = (function () {  let i = 0;  let options = null  const options1 = [    {service: 'Kubernetes Engine', cost: 10.57, discount: 0, total: 10.57},    {service: 'Kubernetes Engine', cost: 5.67, discount: 0, total: 5.67},    {service: 'Kubernetes Engine', cost: 3.42, discount: 0, total: 3.42},  ]  const options2 = [    {service: 'Compute Engine', cost: 10.57, discount: 0, total: 10.57},    {service: 'Compute Engine', cost: 5.67, discount: 0, total: 5.67},    {service: 'Compute Engine', cost: 3.42, discount: 0, total: 3.42},  ]
  return async function () {    options = i % 2 === 0 ? options1 : options2    return await new Promise(function (resolve) {      setTimeout(function () {        resolve(options)        ++i      }, 1000)    })  }})()
