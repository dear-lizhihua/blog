learn_lerna_api

使用 lerna 创建一个 api packages#

git clone git@git.spoton.cn:spoton-cdn/api/api-spoton-cdn-web.gitcd api-spoton-cdn-weblerna init --independentlerna create api-domain -y

每个 package 都安装 axios 和 father-build#

$ lerna add axios$ lerna add father-build --dev

配置 lerna.json#

{  "command": {    "bootstrap": {      "npmClientArgs": ["--no-package-lock"]    }  }}

创建 .fatherrc.js 文件#

export default {  entry: 'lib/index.js',  file: 'api-client',  esm: 'rollup',  umd: {    name: 'apiClient',    sourcemap: false,    minFile: false  }}

配置 package.json#

{  "name": "@api/api-spoton-cdn",  "private": true, // 这个不能少  "version": "1.0.0",  "publishConfig": {    "registry": "https://registry.npmjs.org/"  },  "files": [    "dist"  ],  "main": "dist/api-client.umd.js",  "module": "dist/api-client.esm.js",  "scripts": {    "build": "father-build"  },  "dependencies": {    "axios": "^0.21.1"  },  "devDependencies": {    "father-build": "^1.19.4"  }}

配置 .npmignore#

libnode_modules.fatherrc.js
